import java.util.concurrent.locks.LockSupport;

class Demonstration {
    public static void main( String args[] ) {

        // get a permit
        LockSupport.unpark(Thread.currentThread());
        // consume a permit
        LockSupport.park();

        System.out.println("Main thread exiting");

    }
}
