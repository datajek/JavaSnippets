    private int n;

    public MultithreadedFizzBuzz(int n) {
        this.n = n;
    }

    public void fizz() {
        System.out.print("fizz");
    }

    public void buzz() {
        System.out.print("buzz");        
    }

    public void fizzbuzz() {
        System.out.print("fizzbuzz");
    }

        System.out.print(num);        
    }
